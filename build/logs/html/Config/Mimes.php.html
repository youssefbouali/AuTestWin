512    {
513        $type = trim(strtolower($type), '. ');
514
515        $proposedExtension = trim(strtolower($proposedExtension ?? ''));
516
517        if (
518            $proposedExtension !== ''
519            && array_key_exists($proposedExtension, static::$mimes)
520            && in_array($type, (array) static::$mimes[$proposedExtension], true)
521        ) {
522            // The detected mime type matches with the proposed extension.
523            return $proposedExtension;
524        }

527        // This search is order sensitive!
528        foreach (static::$mimes as $ext => $types) {
529            if (in_array($type, (array) $types, true)) {
530                return $ext;

532        }
